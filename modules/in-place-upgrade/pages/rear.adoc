[source,shell]
----
yum install grub2-efi-x64-modules 
yum install psmisc
----

Install.

----
yum install rear
----

Configure.

----
OUTPUT=ISO
OUTPUT_URL=file:///mnt/rescue_system/
----


Use. 

----
